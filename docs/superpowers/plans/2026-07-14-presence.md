# Presence (Online/Idle/Offline & Last Seen) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Show a live online/idle/offline dot on every `Avatar` that
represents another user (Pantauan, detail Tugas, detail Permintaan, Kelola
Pengguna), a fixed "online" dot on the current user's own avatar
(header/sidebar), and a new read-only `/pengguna/[id]` profile page anyone
can visit to see someone's status + last-seen time.

**Architecture:** A new `presence` table (`userId`, `lastSeenAt`, `status`)
updated by a client heartbeat (every 25s + on activity) from a component
mounted globally in the `(app)` layout. Client-side inactivity (5+ minutes,
tracked via DOM event listeners) flips the reported status to `"idle"`
before the next heartbeat. Staleness (no heartbeat for 3 heartbeat
intervals — tab/browser closed without logging out) is resolved to
`"offline"` at read time, not stored. Pages showing other users' avatars
poll a batched server action every 25s via a small `usePresence` hook;
`Avatar` itself gains a `presence` prop that renders the dot.

**Tech Stack:** Next.js 16 Server Actions, Drizzle migration, no new
dependencies (polling, not push-based — see the approved spec for why).

**Depends on:** [Fitur 2 — Tombol Kembali](2026-07-14-tombol-kembali.md)
(`/pengguna/[id]` uses `SetHeaderBack`), and expects `Avatar` already has
the `src` prop from [Fitur 3](2026-07-14-foto-profil.md) (this plan adds a
second, independent optional prop — `presence` — alongside it; if Fitur 3
hasn't been implemented yet, skip the `src=` parts of this plan's JSX
snippets and it still applies cleanly on top of the original `Avatar`).

## Global Constraints

- Heartbeat interval: 25 seconds. Staleness threshold: 3× that (75s) — past
  that, a user reads as offline regardless of stored status.
- Idle threshold: 5 minutes of no `mousemove`/`keydown`/`touchstart`/`scroll`.
- Own avatar (header, sidebar) always shows a **fixed** `"online"` dot —
  no polling needed, it's definitionally true whenever the page is loaded
  (matches the approved spec's literal wording; idle nuance only applies to
  how *others* see you, not your own header icon).
- Explicit logout immediately deletes the user's presence row (instant
  offline), rather than waiting for staleness.
- Read functions live in `src/lib/data/presence.ts` (server-only, called
  directly from Server Components); the client-facing RPC wrapper and the
  write path live in `src/lib/actions/presence.ts` — matching this
  project's existing `data/` vs `actions/` split (see `CLAUDE.md`).

```ts
export type PresenceStatus = "online" | "idle" | "offline";

export const HEARTBEAT_INTERVAL_MS = 25_000;
export const IDLE_AFTER_MS = 5 * 60 * 1000;
const STALE_AFTER_MS = HEARTBEAT_INTERVAL_MS * 3;

/**
 * Status presence yang ditampilkan, dihitung saat baca (bukan disimpan
 * sebagai enum 3 nilai): heartbeat basi -> offline apa pun status
 * tersimpan; kalau masih segar, pakai status yang dilaporkan client.
 */
export function resolvePresenceStatus(
  lastSeenAt: Date | null,
  reportedStatus: string | null,
  now: Date = new Date(),
): PresenceStatus {
  if (!lastSeenAt) return "offline";
  const age = now.getTime() - lastSeenAt.getTime();
  if (age > STALE_AFTER_MS) return "offline";
  return reportedStatus === "idle" ? "idle" : "online";
}
```

- [ ] **Step 1: Write the module**

Create `src/lib/data/presence.ts`:

```ts
import { inArray } from "drizzle-orm";
import { db } from "@/db";
import { presence } from "@/db/schema";
import { resolvePresenceStatus, type PresenceStatus } from "@/lib/presence";

export type PresenceInfo = { status: PresenceStatus; lastSeenAt: Date | null };

export async function getPresenceForUsers(userIds: string[]): Promise<Record<string, PresenceInfo>> {
  if (userIds.length === 0) return {};
  const rows = await db.query.presence.findMany({ where: inArray(presence.userId, userIds) });
  const byId = new Map(rows.map((r) => [r.userId, r]));
  const now = new Date();
  const result: Record<string, PresenceInfo> = {};
  for (const id of userIds) {
    const row = byId.get(id);
    result[id] = {
      status: resolvePresenceStatus(row?.lastSeenAt ?? null, row?.status ?? null, now),
      lastSeenAt: row?.lastSeenAt ?? null,
    };
  }
  return result;
}
```

- [ ] **Step 1: Write the actions**

Create `src/lib/actions/presence.ts`:

```ts
"use server";

import { eq } from "drizzle-orm";
import { db } from "@/db";
import { presence } from "@/db/schema";
import { requireUser } from "@/lib/auth";
import { getPresenceForUsers } from "@/lib/data/presence";
import type { PresenceStatus } from "@/lib/presence";

export async function heartbeatAction(clientStatus: "online" | "idle"): Promise<void> {
  const me = await requireUser();
  await db
    .insert(presence)
    .values({ userId: me.id, status: clientStatus, lastSeenAt: new Date() })
    .onConflictDoUpdate({
      target: presence.userId,
      set: { status: clientStatus, lastSeenAt: new Date() },
    });
}

/** Dipakai client (usePresence) untuk polling status user lain — cuma butuh status, bukan lastSeenAt mentah. */
export async function getPresenceStatusesAction(
  userIds: string[],
): Promise<Record<string, PresenceStatus>> {
  await requireUser();
  const info = await getPresenceForUsers(userIds);
  const statuses: Record<string, PresenceStatus> = {};
  for (const id of userIds) statuses[id] = info[id]?.status ?? "offline";
  return statuses;
}

export async function clearPresenceAction(userId: string): Promise<void> {
  await db.delete(presence).where(eq(presence.userId, userId));
}
```

- [ ] **Step 1: Write the component**

Create `src/components/app-shell/presence-heartbeat.tsx`:

```tsx
"use client";

import { useEffect, useRef } from "react";
import { heartbeatAction } from "@/lib/actions/presence";
import { HEARTBEAT_INTERVAL_MS, IDLE_AFTER_MS } from "@/lib/presence";

const ACTIVITY_EVENTS = ["mousemove", "keydown", "touchstart", "scroll"] as const;

/** Tidak merender apa pun — mengirim heartbeat presence tiap ~25 detik. */
export function PresenceHeartbeat() {
  const lastActivityRef = useRef(Date.now());

  useEffect(() => {
    function markActive() {
      lastActivityRef.current = Date.now();
    }
    for (const event of ACTIVITY_EVENTS) {
      window.addEventListener(event, markActive, { passive: true });
    }

    function sendHeartbeat() {
      const idleFor = Date.now() - lastActivityRef.current;
      void heartbeatAction(idleFor >= IDLE_AFTER_MS ? "idle" : "online");
    }

    sendHeartbeat();
    const interval = setInterval(sendHeartbeat, HEARTBEAT_INTERVAL_MS);

    function handleVisibility() {
      if (document.visibilityState === "visible") sendHeartbeat();
    }
    document.addEventListener("visibilitychange", handleVisibility);

    return () => {
      for (const event of ACTIVITY_EVENTS) window.removeEventListener(event, markActive);
      document.removeEventListener("visibilitychange", handleVisibility);
      clearInterval(interval);
    };
  }, []);

  return null;
}
```

- [ ] **Step 1: Write the hook**

Create `src/lib/use-presence.ts`:

```ts
"use client";

import { useEffect, useState } from "react";
import { getPresenceStatusesAction } from "@/lib/actions/presence";
import { HEARTBEAT_INTERVAL_MS, type PresenceStatus } from "@/lib/presence";

/** Polling batched presence untuk daftar userId yang sedang tampil di layar. */
export function usePresence(userIds: string[]): Record<string, PresenceStatus> {
  const key = userIds.join(",");
  const [statuses, setStatuses] = useState<Record<string, PresenceStatus>>({});

  useEffect(() => {
    if (!key) return;
    let cancelled = false;

    async function poll() {
      const result = await getPresenceStatusesAction(key.split(","));
      if (!cancelled) setStatuses(result);
    }

    poll();
    const interval = setInterval(poll, HEARTBEAT_INTERVAL_MS);
    return () => {
      cancelled = true;
      clearInterval(interval);
    };
  }, [key]);

  return statuses;
}
```

### Task 13: New `/pengguna/[id]` profile-view page

**Files:**
- Create: `src/app/(app)/pengguna/[id]/page.tsx`

**Interfaces:**
- Consumes: `getUserById` (existing), `getPresenceForUsers` (Task 3),
  `PresenceAvatar` (Task 7), `SetHeaderBack` (Fitur 2).

- [ ] **Step 1: Write the page**

Create `src/app/(app)/pengguna/[id]/page.tsx`:

```tsx
import { notFound } from "next/navigation";
import { requireUser } from "@/lib/auth";
import { getUserById } from "@/lib/data/users";
import { getPresenceForUsers } from "@/lib/data/presence";
import { PresenceAvatar } from "@/components/presence/presence-avatar";
import { SetHeaderBack } from "@/components/app-shell/header-back";
import { timeAgo } from "@/lib/format";

export default async function PenggunaDetailPage({ params }: { params: Promise<{ id: string }> }) {
  await requireUser();
  const { id } = await params;
  const user = await getUserById(id);
  if (!user) notFound();

  const { status, lastSeenAt } = (await getPresenceForUsers([id]))[id];

  return (
    <div className="space-y-4 pb-4 lg:mx-auto lg:max-w-xl">
      <SetHeaderBack title="Profil Pengguna" fallbackHref="/beranda" />
      <div className="flex items-center gap-4 rounded-2xl border border-slate-200 bg-white p-5 shadow-sm">
        <PresenceAvatar userId={user.id} name={user.name} src={user.avatarUrl} size="lg" />
        <div className="min-w-0">
          <p className="truncate text-lg font-bold text-slate-900">{user.name}</p>
          <p className="text-sm text-slate-500">{user.position.name}</p>
          <p className="mt-1.5 text-sm font-medium">
            {status === "online" && <span className="text-emerald-600">Online sekarang</span>}
            {status === "idle" && <span className="text-amber-600">Idle</span>}
            {status === "offline" && (
              <span className="text-slate-500">
                {lastSeenAt ? `Terakhir dilihat: ${timeAgo(lastSeenAt)}` : "Belum pernah online"}
              </span>
            )}
          </p>
        </div>
      </div>
    </div>
  );
}
```

(This route uses `SetHeaderBack` from Fitur 2 — if that plan hasn't run
yet, drop the import and that line, and add back a plain `<h1>` title
instead, since there'd be no header back-button slot to register into.)
